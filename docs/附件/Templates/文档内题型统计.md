
> [!summary] 本文档题型统计
> ```dataviewjs
> const file = app.vault.getAbstractFileByPath(dv.current().file.path);
> const content = await app.vault.cachedRead(file);
> 
> const a = (content.match(/> \[!tip\]/g) || []).length;
> const b = (content.match(/> \[!example\]/g) || []).length;
> const c = (content.match(/> \[!note\]/g) || []).length;
> 
> // 只有大于0才显示
> if (a > 0) dv.paragraph(`- **名词解释**：${a} 个`);
> if (b > 0) dv.paragraph(`- **简答**：${b} 个`);
> if (c > 0) dv.paragraph(`- **论述**：${c} 个`);
> 
> // 如果全都是0，显示提示；否则显示总计
> if (a + b + c === 0) dv.paragraph(`<small>本文档暂无题型</small>`);
> else dv.paragraph(`<small>当前文档总计：${a+b+c} 个</small>`);
> ```
